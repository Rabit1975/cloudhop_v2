### Basic Nebula Background

This example demonstrates a simple usage of the NebulaBackground component, displaying a welcoming message within the nebula effect.

```tsx
import { NebulaBackground } from './nebula-background.js';
import React from 'react';

const BasicUsage = () => {
  return (
    <div style={{ height: '500px', width: '100%' }}>
      <NebulaBackground>
        <div style={{ 
          display: 'flex', 
          flexDirection: 'column',
          alignItems: 'center', 
          justifyContent: 'center', 
          height: '100%',
          minHeight: '400px',
          color: '#E2E8F0', // text-primary
          textAlign: 'center'
        }}>
          <h1 style={{ 
            fontSize: '3rem', // display-medium
            margin: '0 0 1rem 0' // spacing-medium
          }}>
            CloudHop Social
          </h1>
          <p style={{ 
            fontSize: '1.25rem', // body-large
            maxWidth: '600px',
            color: '#CBD5E1' // text-secondary
          }}>
            Experience the next generation of social interaction with our immersive nebula environment.
          </p>
        </div>
      </NebulaBackground>
    </div>
  );
};
export default BasicUsage;
```

### Social Hub Login Interface

This example shows the NebulaBackground used for a login interface, with the intensified prop to make the nebula effect more prominent, suitable for focused sections like authentication.

```tsx
import { NebulaBackground } from './nebula-background.js';
import React from 'react';

const LoginInterface = () => {
  return (
    <div style={{ height: '100vh', width: '100%' }}>
      <NebulaBackground intensified>
        <div style={{ 
          display: 'flex', 
          alignItems: 'center', 
          justifyContent: 'center', 
          height: '100%',
          width: '100%'
        }}>
          <div style={{
            width: '100%',
            maxWidth: '400px',
            padding: '48px',
            backgroundColor: 'rgba(15, 23, 42, 0.6)',
            backdropFilter: 'blur(16px)', // blur-medium
            borderRadius: '12px', // radius-large
            border: '1px solid rgba(71, 85, 105, 0.5)', // border-subtle
            boxShadow: '0 10px 15px -3px rgba(0,0,0,0.1), 0 4px 6px -2px rgba(0,0,0,0.05)', // shadows-large
            display: 'flex',
            flexDirection: 'column',
            gap: '24px',
            color: '#E2E8F0' // text-primary
          }}>
            <div style={{ textAlign: 'center' }}>
              <div style={{ 
                width: '64px', 
                height: '64px', 
                backgroundColor: '#3B82F6', // primary-default
                borderRadius: '50%',
                margin: '0 auto 16px',
                display: 'flex',
                alignItems: 'center',
                justifyContent: 'center',
                fontSize: '24px',
                fontWeight: 'bold',
                boxShadow: '0 0 15px rgba(59, 130, 246, 0.6)' // glow shadow
              }}>
                CH
              </div>
              <h2 style={{ margin: 0, fontSize: '1.875rem' }}>Welcome Back</h2> {/* heading-h3 */}
              <p style={{ color: '#CBD5E1', marginTop: '8px' }}>Enter the nebula to continue</p> {/* text-secondary */}
            </div>
            
            <div style={{ display: 'flex', flexDirection: 'column', gap: '16px' }}>
              <input 
                type="text"

                  fontSize: '16px'

              />
              <input 
                type="password" 
                placeholder="Password"

                  padding: '12px 16px',

                  backgroundColor: 'rgba(30, 41, 59, 0.7)',
                  color: '#E2E8F0',

                fontSize: '16px',

                Log In
              </button>
            </div>
          </div>
        </div>
      </NebulaBackground>
    </div>
  );
};
export default LoginInterface;

### HopHub Chat Interface

This example illustrates how the NebulaBackground can be integrated into a complex layout, such as a chat application interface, providing a consistent visual theme.

```tsx
import { NebulaBackground } from './nebula-background.js';
import React from 'react';

const ChatInterface = () => {
  return (
    <div style={{ height: '600px', width: '100%', position: 'relative', overflow: 'hidden' }}>
      <NebulaBackground style={{ position: 'absolute', height: '100%', minHeight: 'unset' }}>
        <div style={{ 
          display: 'flex', 
          height: '100%', 
          width: '100%',
          color: '#E2E8F0' // text-primary
        }}>
          {/* Server Sidebar */}
          <div style={{ 
            width: '72px', 
            backgroundColor: 'rgba(15, 23, 42, 0.8)', 
            backdropFilter: 'blur(16px)', // blur-medium
            display: 'flex', 
            flexDirection: 'column', 
            alignItems: 'center', 
            padding: '16px 0',
            gap: '16px',
            borderRight: '1px solid rgba(71, 85, 105, 0.5)', // border-subtle
            zIndex: 2
          }}>
            {[1, 2, 3].map((i) => (
              <div key={i} style={{ 
                width: '48px', 
                height: '48px', 
                borderRadius: '50%', 
                backgroundColor: i === 1 ? '#3B82F6' : 'rgba(255,255,255,0.1)', // primary-default
                display: 'flex',
                alignItems: 'center',
                justifyContent: 'center',
                cursor: 'pointer',
                boxShadow: i === 1 ? '0 0 10px rgba(59, 130, 246, 0.4)' : 'none', // glow shadow
                transition: 'all 0.2s ease',
                fontSize: '14px',
                fontWeight: 'bold'
              }}>
                {i === 1 ? 'CH' : `S${i}`}
              </div>
            ))}
          </div>
          {/* Channel Sidebar */}
          <div style={{ 
            width: '240px', 
            backgroundColor: 'rgba(30, 41, 59, 0.4)', 
            padding: '24px',
            display: 'flex',
            flexDirection: 'column',
            backdropFilter: 'blur(8px)', // blur-small
            borderRight: '1px solid rgba(71, 85, 105, 0.5)', // border-subtle
            zIndex: 1
          }}>
            <div style={{ 
              fontSize: '1.5rem', // heading-h5
              fontWeight: 'bold', 
              marginBottom: '24px',
              color: '#E2E8F0' // text-primary
            }}>
              CloudHop HQ
            </div>
            
            <div style={{ marginBottom: '24px' }}>
              <div style={{ 
                textTransform: 'uppercase', 
                fontSize: '11px', 
                color: '#CBD5E1', // text-secondary
                fontWeight: 'bold',
                marginBottom: '12px'
              }}>
                HopHub Channels
              </div>
              {['# general', '# announcements', '# random', '# music-lounge'].map(channel => (
                <div key={channel} style={{ 
                  padding: '8px 12px', 
                  borderRadius: '4px',
                  cursor: 'pointer',
                  color: channel === '# general' ? '#E2E8F0' : '#CBD5E1', // text-primary : text-secondary
                  backgroundColor: channel === '# general' ? 'rgba(255,255,255,0.1)' : 'transparent',
                  marginBottom: '4px',
                  fontSize: '14px'
                }}>
                  {channel}
                </div>
              ))}
            </div>
          </div>
          {/* Main Content Area */}
          <div style={{ 
            flex: 1, 
            display: 'flex', 
            flexDirection: 'column',
            position: 'relative'
          }}>
            {/* Header */}
            <div style={{ 
              height: '64px', 
              borderBottom: '1px solid rgba(71, 85, 105, 0.5)', // border-subtle
              display: 'flex',
              alignItems: 'center',
              padding: '0 24px',
              backgroundColor: 'rgba(15, 23, 42, 0.2)',
              backdropFilter: 'blur(8px)' // blur-small
            }}>
              <span style={{ fontSize: '18px', fontWeight: 'bold' }}># general</span>
              <span style={{ marginLeft: '16px', fontSize: '14px', color: '#CBD5E1' }}> {/* text-secondary */}
                Welcome to the nebula...
              </span>
            </div>
            {/* Chat Messages */}
            <div style={{ flex: 1, padding: '24px', overflowY: 'auto' }}>
               {[
                { user: 'Alice', time: '10:42 AM', text: 'Has anyone seen the new nebula theme update?' },
                { user: 'Bob', time: '10:45 AM', text: 'Yes! The gradients are so smooth. Really gives that "Space" vibe.' },
                { user: 'Charlie', time: '10:48 AM', text: 'I love the new "Glow" shadow effects on the active server icons.' }
              ].map((msg, idx) => (
                <div key={idx} style={{ display: 'flex', gap: '16px', marginBottom: '24px' }}>
                  <div style={{ 
                    width: '40px', 
                    height: '40px', 
                    borderRadius: '50%', 
                    backgroundColor: `hsl(${220 + (idx * 40)}, 70%, 60%)`,
                    flexShrink: 0
                  }} />
                  <div>
                    <div style={{ display: 'flex', alignItems: 'baseline', gap: '8px', marginBottom: '4px' }}>
                      <span style={{ fontWeight: 'bold', color: '#3B82F6' }}>{msg.user}</span> {/* primary-default */}
                      <span style={{ fontSize: '12px', color: '#CBD5E1' }}>{msg.time}</span> {/* text-secondary */}
                    </div>
                    <div style={{ lineHeight: '1.5', color: '#F8FAFC' }}>{msg.text}</div> {/* text-default */}
                  </div>
                </div>
              ))}
            </div>
            {/* Input Area */}
            <div style={{ padding: '24px' }}>
              <div style={{ 
                backgroundColor: 'rgba(0,0,0,0.3)', 
                padding: '16px', 
                borderRadius: '8px',
                color: '#CBD5E1', // text-secondary
                border: '1px solid rgba(71, 85, 105, 0.5)' // border-subtle
              }}>
                Message #general...
              </div>
            </div>
          </div>
        </div>
      </NebulaBackground>
    </div>
  );
};
export default ChatInterface;
