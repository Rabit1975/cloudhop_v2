A component for a Meeting entity, supporting serialization and de-serialization of Meeting objects. It can be used from both backend and frontend applications.

### Creating a Meeting object

This example shows how to create a new meeting object using the static from method

```ts
import { Meeting } from './meeting.js';
import { PlainMeeting } from './meeting-type.js';
const now = new Date();
const oneHourLater = new Date(now.getTime() + 3600 * 1000);
const plainMeeting: PlainMeeting = {
  id: 'meet-123',
  topic: 'Project Sync Up',
  description: 'Discuss Q3 project milestones and next steps',
  startTime: now.toISOString(),
  endTime: oneHourLater.toISOString(),
  hostId: 'user-h123',
  participants: [
    { userId: 'user-h123', role: 'host' },
    { userId: 'user-a456', role: 'attendee' },
    { userId: 'user-b789', role: 'attendee' },
  ],
  status: 'scheduled',
  accessCode: 'P-R0J3CT',
  joinUrl: 'https://cloudhop.com/meet/projectsync',
  createdAt: now.toISOString(),
};
const meeting = Meeting.from(plainMeeting);
console.log(meeting.topic);
// Expected output: Project Sync Up
```
### Accessing meeting details

This example demonstrates how to access various properties and computed values of a meeting instance
```ts
import { Meeting } from './meeting.js';
import { PlainMeeting } from './meeting-type.js';
const now = new Date();
const oneHourLater = new Date(now.getTime() + 3600 * 1000);
const plainMeetingData: PlainMeeting = {
  id: 'meet-456',
  topic: 'Client Demo',
  description: 'Showcase the new features to the client',
  startTime: now.toISOString(),
  endTime: oneHourLater.toISOString(),
  hostId: 'user-host2',
  participants: [{ userId: 'user-host2', role: 'host' }],
  status: 'live',
  joinUrl: 'https://cloudhop.com/meet/clientdemo',
  createdAt: now.toISOString(),
  accessCode: 'DEMO4ME',
};
const clientMeeting = Meeting.from(plainMeetingData);
console.log(clientMeeting.title);
// Expected output: Client Demo
console.log(clientMeeting.isScheduled);
// Expected output: false
console.log(clientMeeting.meetingRoomId);
// Expected output: DEMO4ME
console.log(clientMeeting.meetingLink);
// Expected output: https://cloudhop.com/meet/clientdemo
const plainObject = clientMeeting.toObject();
console.log(plainObject.status);
// Expected output: live
```
### Using mock meeting data

This example illustrates how to generate mock meeting data for development and testing purposes
```ts
import { mockMeeting, mockMeetings } from './meeting.mock.js';
// Generate a single mock meeting
const singleMeeting = mockMeeting({ topic: 'Daily Standup' });
console.log(singleMeeting.topic);
// Expected output: Daily Standup (or other default topic if not overridden)
// Generate multiple mock meetings
const threeMeetings = mockMeetings(3);
console.log(`Generated ${threeMeetings.length} meetings`);
// Expected output: Generated 3 meetings
