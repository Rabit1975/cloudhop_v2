A React hook for managing meeting data. It exposes independent hooks for fetching a single meeting, listing meetings, creating new meetings, updating existing meetings, and deleting meetings using GraphQL mutations. It also handles loading and error states.

### Fetching a single meeting

This example demonstrates how to fetch and display details for a specific meeting using its ID. It simulates data retrieval and shows how to render the meeting topic or a loading and error message.

```tsx
const MeetingDetailComponent = () => {
  // Assume a meeting ID is available
  const meetingId = 'meeting-abc-123';
  // Mock data for demonstration purposes
  const mockMeetingData = {
    id: meetingId,
    topic: 'Team Sync Meeting',
    description: 'Weekly team synchronization meeting',
    startTime: '2023-10-27T10:00:00Z',
    endTime: '2023-10-27T11:00:00Z',
    hostId: 'user-456',
    status: 'SCHEDULED',
    accessCode: '12345',
    joinUrl: 'https://example.com/join/meeting-abc-123',
  };
  const { meeting, loading, error } = useMeeting({
    meetingOptions: { meetingId },
    mockData: mockMeetingData
  });
  if (loading) {
    return <p>Loading meeting details...</p>;
  }
  if (error) {
    return <p>Error loading meeting: {error}</p>;
  }
  if (!meeting) {
    return <p>No meeting found.</p>;
  }
  return (
    <div>
      <h3>Meeting Topic: {meeting.topic}</h3>
      <p>Description: {meeting.description}</p>
      <p>Start Time: {meeting.startTime}</p>
    </div>
  );
};
```
### Listing meetings with filters
This example shows how to list multiple meetings and apply filters, such as by participant user ID. It demonstrates displaying a list of meeting topics or showing loading and error states.
```tsx
const MeetingListComponent = () => {
  // Assume a participant user ID for filtering
  const participantUserId = 'user-789';
  // Mock data for demonstration purposes
  const mockMeetingsData = [
    {
      id: 'meeting-1',
      topic: 'Project Review',
      description: 'Review of current project status',
      startTime: '2023-10-26T14:00:00Z',
      endTime: '2023-10-26T15:00:00Z',
      hostId: 'user-789',
      status: 'COMPLETED',
      accessCode: '54321',
      joinUrl: 'https://example.com/join/meeting-1',
    },
    {
      id: 'meeting-2',
      topic: 'New Feature Brainstorm',
      description: 'Brainstorming session for upcoming feature',
      startTime: '2023-10-28T09:30:00Z',
      endTime: '2023-10-28T10:30:00Z',
      hostId: 'user-101',
      status: 'SCHEDULED',
      accessCode: '98765',
      joinUrl: 'https://example.com/join/meeting-2',
    },
  ];
  const { meetings, loading, error } = useMeetingList({
    listOptions: { participantUserId: participantUserId, limit: 10 },
    mockData: mockMeetingsData
  });
  if (loading) {
    return <p>Loading meetings...</p>;
  }
  if (error) {
    return <p>Error loading meetings: {error}</p>;
  }
  if (meetings.length === 0) {
    return <p>No meetings found for this participant.</p>;
  }
  return (
    <div>
      <h3>Meetings for Participant {participantUserId}</h3>
      <ul>
        {meetings.map((meeting) => (
          <li key={meeting.id}>
            {meeting.topic} (Status: {meeting.status})
          </li>
        ))}
      </ul>
    </div>
  );
};
```
### Creating a new meeting
This example illustrates how to create a new meeting. It includes a button to trigger the creation process and displays the newly created meeting topic or an error message.
```tsx
const CreateMeetingComponent = () => {
  const [createMeeting, { meeting, loading, error }] = useCreateMeeting();
  const handleCreateMeeting = async () => {
    try {
      await createMeeting({
        topic: 'New Project Kickoff',
        description: 'Kickoff meeting for the new project phase',
        startTime: '2023-11-01T10:00:00Z',
        endTime: '2023-11-01T11:00:00Z',
        participantUserIds: ['user-123', 'user-456'],
      });
    } catch (e) {
      console.error('Failed to initiate meeting creation:', e);
    }
  };
  return (
    <div>
      <h3>Create a New Meeting</h3>
      <button onClick={handleCreateMeeting} disabled={loading}>
        {loading ? 'Creating...' : 'Create Meeting'}
      </button>
      {error && <p>Error creating meeting: {error}</p>}
      {meeting && <p>Meeting created successfully. Topic: {meeting.topic}</p>}
    </div>
  );
};
