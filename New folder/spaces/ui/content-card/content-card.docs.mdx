### Document Content Card

This example shows a content card displaying a document type with a title and a body excerpt.

```tsx
<div style={{ padding: '32px', maxWidth: '340px' }}>
  {(() => {
    // Helper to create a mock content entity for documentation purposes
    const createMockContent = (data) => ({
      toObject: () => ({
        id: 'mock-doc-1',
        createdAt: new Date().toISOString(),
        updatedAt: new Date().toISOString(),
        ...data,
      }),
    });
    const content = createMockContent({
      title: 'Nebula Protocol Guidelines',
      body: 'A comprehensive document outlining the communication protocols within the CloudHop ecosystem. Essential for all new Space administrators.',
      type: 'document',
    });
    return <ContentCard content={content} href="/spaces/docs/1" />;
  })()}
</div>
```

### Visual Content Card

This example demonstrates a content card used for visual assets, featuring a title, body, and a custom thumbnail image.

```tsx
<div style={{ padding: '32px', maxWidth: '340px' }}>
  {(() => {
    // Helper to create a mock content entity for documentation purposes
    const createMockContent = (data) => ({
      toObject: () => ({
        id: 'mock-visual-1',
        createdAt: new Date().toISOString(),
        updatedAt: new Date().toISOString(),
        ...data,
      }),
    });
    const SAMPLE_IMAGE = "https://storage.googleapis.com/bit-generated-images/images/image_abstract_digital_content_card__0_1770833866309.png";
    const content = createMockContent({
      title: 'Space Station Concepts',
      body: 'Early concept art sketches for the new orbital hub environment in GameHub.',
      type: 'image',
    });
    return <ContentCard content={content} href="/spaces/assets/2" thumbnailUrl={SAMPLE_IMAGE} />;
  })()}
</div>
```

### Multiple Content Cards in a Grid

This example illustrates how to display several content cards in a responsive grid layout, useful for content listings.

```tsx
<div style={{
  padding: '32px',
  display: 'grid',
  gridTemplateColumns: 'repeat(auto-fit, minmax(280px, 1fr))',
  gap: '24px',
  backgroundColor: 'var(--colors-surface-background)'
}}>
  {(() => {
    // Helper to create a mock content entity for documentation purposes
    const createMockContent = (data) => ({
      toObject: () => ({
        id: `mock-${data.title.replace(/\s/g, '-').toLowerCase()}`,
        createdAt: new Date().toISOString(),
        updatedAt: new Date().toISOString(),
        ...data,
      }),
    });
    const contentList = [
      createMockContent({
        title: 'Q4 Roadmap',
        type: 'document',
        body: 'Strategic goals and milestones for the upcoming quarter.',
      }),
      createMockContent({
        title: 'Interface Mockups',
        type: 'design',
        body: 'High-fidelity designs for the new chat interface.',
      }),
      createMockContent({
        title: 'User Feedback',
        type: 'document',
        body: 'Aggregated feedback from the beta testing phase.',
      }),
    ];
    return (
      contentList.map((item, idx) => (
        <ContentCard
          key={item.toObject().id || idx}
          content={item}
          href={`/spaces/content/${idx}`}
        />
      ))
    );
  })()}
</div>
