### Basic Usage

A basic rendering of the forgot password component with default behavior.

```tsx
<ForgotPassword />
```

### With a custom submit handler
Demonstrates how to provide a custom function to handle the email submission.

```tsx
function CustomForgotPassword() {
  const [submittedEmail, setSubmittedEmail] = useState('');
  const handleSubmit = (email: string) => {
    console.log('Submitted email:', email);
    setSubmittedEmail(email);
    alert(`Password reset requested for ${email}. Check console for details.`);
  };
  return (
    <div>
      <ForgotPassword onSubmit={handleSubmit} />
      {submittedEmail && <p>Last submitted email: {submittedEmail}</p>}
    </div>
  );
}
<CustomForgotPassword />
